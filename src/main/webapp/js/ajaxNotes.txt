In .JS file:

function deleteSelected(namespace, href) {

    // Get selected elements from posts table
    var postsSelected = new Array();
    $("#" + namespace + "_posts").children("tbody").children().each(function() {
        $row = $(this);
        var checkbox = $row.find("input:checkbox");
        if (checkbox.is(":checked")) {
            postsSelected.push(checkbox.val())
        }
    });

    alert("Rows detected: " + postsSelected);

    $.ajax({
        type: "POST",
        url: href + "&selected=" + postsSelected,
        cache: false,
        dataType: "text",
        success: function(data)
        {
            alert("Ajax call ok !");
        },
        error: function(XMLHttpRequest, textStatus, errorThrown)
        {
            alert("Ajax call KO !!");
        }
    });

}
